一个创建模拟数据的存储过程


DROP PROCEDURE IF EXISTS `insertStubData`;
DELIMITER $$
CREATE PROCEDURE `insertStubData`()
BEGIN
    DECLARE i int unsigned DEFAULT 0;
    WHILE i < 30000 DO
        SET i = i+1;
        INSERT INTO `sso_login`(`id`,
        `phone_nbr`,
        `open_id`,
        `project_from`,
        `task_id`)
        VALUES
        (null,CONCAT("133",i),i,"TW",10);
    END WHILE;
END $$
DELIMITER ;

CALL insertStubData();
